struct FIFO * create(struct FIFO *tail, int x) {
	struct FIFO *n;    // указатель на новую структуру
	n = (struct FIFO *)malloc(sizeof(struct FIFO)); // выделяем память
	n->next = NULL;
	n->data = x;
	if (tail == NULL)	
	{
		tail= n;
	}
	else
	{
		tail->next=n;
		tail=n;
	}
	return tail;
}

void print(struct FIFO *head){
	struct FIFO *p = head;
	while (p != NULL) {      
		printf("%2d ---> ", p->data);
		p = p->next;       // продвижение по списку
	}
	printf("\n");
}

void pop(struct FIFO **head, struct FIFO ** tail)
{
	if (*head != NULL)
	{
		*head = (*head)->next;
		
	}
	else 
	{
		*tail=NULL;
	}
